QUERY: 1000_AfiladoCI2Nombre
  SQL: PARAMETERS pCI Long;
SELECT SP_Afiliado.CI AS CI, [SP_Afiliado].[Nombres] & " " & [SP_Afiliado].[Apellido1] & IIf([SP_Afiliado].[Apellido2] & ""<>""," " ,"") & [SP_Afiliado].[Apellido2] AS DescAfiliado, SP_Afiliado.Direccion, SP_Afiliado.Telefono
FROM SP_Afiliado
WHERE (((SP_Afiliado.CI)=[pCI]));


QUERY: 1000_AfiliadoImpLiquidoxMes
  SQL: SELECT SP_ImpLiquido.CI, SP_ImpLiquido.Anio, SP_ImpLiquido.Mes, Sum(SP_ImpLiquido.Importe) AS Importe, First(SP_ImpLiquido.AnioMes) AS AnioMes
FROM SP_ImpLiquido INNER JOIN SP_Trabaja ON (SP_ImpLiquido.Fechaingreso = SP_Trabaja.FechaIngreso) AND (SP_ImpLiquido.CodEmpresa = SP_Trabaja.CodEmpresa) AND (SP_ImpLiquido.CI = SP_Trabaja.CI)
WHERE (((SP_Trabaja.FechaBaja) Is Null Or (SP_Trabaja.FechaBaja)>Date()) AND ((SP_Trabaja.FechaIngreso)<=Date()))
GROUP BY SP_ImpLiquido.CI, SP_ImpLiquido.Anio, SP_ImpLiquido.Mes;


QUERY: 1000_AfiliadoPromedioMeses
  SQL: PARAMETERS pCI Long, pAnioMesIni Long, pAnioMesFin Long, pMeses Short;
SELECT Sum([1000_AfiliadoImpLiquidoxMes].Importe)/[pMeses] AS Importe
FROM 1000_AfiliadoImpLiquidoxMes
WHERE ((([1000_AfiliadoImpLiquidoxMes].CI)=[pCI]) AND (([1000_AfiliadoImpLiquidoxMes].AnioMes) Between [pAnioMesIni] And [pAnioMesFin]));

QUERY: 1001_AfiliadoImpLiquidoxCI
  SQL: PARAMETERS pCI Long;
SELECT SP_ImpLiquido.CI, SP_ImpLiquido.CodEmpresa, SP_Empresa.Nombre AS DescEmpresa, SP_ImpLiquido.Anio, SP_ImpLiquido.Mes, SP_ImpLiquido.Importe, SP_ImpLiquido.AnioMes
