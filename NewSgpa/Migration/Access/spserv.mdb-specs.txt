=== QUERIES ===
QUERY: Cobros
  SQL: SELECT SP_RetencionPago.IDPrestamo, Sum(SP_RetencionPago.Importe) AS Importe
FROM SP_RetencionPago
GROUP BY SP_RetencionPago.IDPrestamo;


QUERY: envios
  SQL: SELECT SP_Retencion.IDPrestamo, Sum(SP_Retencion.Importe) AS Importe
FROM SP_Retencion
GROUP BY SP_Retencion.IDPrestamo;


QUERY: Rs_PrestamoInteresAmortizacion
  SQL: SELECT SP_CuadroAmortizacion.IDPrestamo, Sum(SP_CuadroAmortizacion.Interes) AS Interes, Sum(SP_CuadroAmortizacion.Amortizacion) AS Amortizacion
FROM SP_CuadroAmortizacion
GROUP BY SP_CuadroAmortizacion.IDPrestamo;


QUERY: rsCtaCteRet
  SQL: SELECT envios.IDPrestamo, envios.Importe, iif(isnull(Cobros.Importe), 0, Cobros.Importe) AS cobros, Iif(abs(envios.Importe-cobros)<1, 0, envios.Importe-cobros) AS saldo
FROM envios LEFT JOIN Cobros ON envios.IDPrestamo = Cobros.IDPrestamo;


QUERY: rsDatos
  SQL: SELECT rsFacturaEstado.Mes, Count(*) AS Cantidad, rsFacturaEstado.CodMoneda, Sum(rsFacturaEstado.Importe) AS Importe
FROM rsFacturaEstado
GROUP BY rsFacturaEstado.Mes, rsFacturaEstado.CodMoneda;


QUERY: rsDatos_Tabla de referencias cruzadas
  SQL: TRANSFORM Sum([Cantidad]) AS [El Valor]
SELECT [IdPrestamo], Sum([Cantidad]) AS [Total de Cantidad]
FROM rsDatos
GROUP BY [IdPrestamo]
PIVOT [Mes];


QUERY: rsFacturaEstado
  SQL: SELECT SP_Factura.IdPrestamo, SP_Factura.IDFactura, Format(SP_Factura.FechaVencimiento,"yyyymm") AS Mes, SP_Prestamo.CodMoneda, SP_Prestamo.Importe
FROM SP_PrestamoEstado INNER JOIN (SP_Prestamo INNER JOIN SP_Factura ON SP_Prestamo.IDPrestamo = SP_Factura.IdPrestamo) ON SP_PrestamoEstado.CodPrestamoEstado = SP_Prestamo.CodPrestamoEstado
WHERE (((SP_Factura.CodFacturaEstado)<>"anu") AND ((SP_PrestamoEstado.Fin)=False) AND ((SP_PrestamoEstado.CodPrestamoEstado)<>"anu") AND ((SP_Factura.FechaVencimiento)>#11/30/2003#));
