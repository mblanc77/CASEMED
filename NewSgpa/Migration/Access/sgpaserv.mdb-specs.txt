  SQL: UPDATE tExcel INNER JOIN (SubsidioCabezal INNER JOIN (SubsidioEnfermedad INNER JOIN Certificacion ON SubsidioEnfermedad.FechaIni = Certificacion.FechaIni) ON (SubsidioCabezal.IdSubsidio = SubsidioEnfermedad.IdSubsidio) AND (SubsidioCabezal.CI = Certificacion.CI)) ON tExcel.CI = Certificacion.CI SET SubsidioEnfermedad.FechaFin = [Certificacion].[FechaFin]
WHERE (((SubsidioEnfermedad.FechaFin)<>[Certificacion].[FechaFin]) AND ((SubsidioCabezal.Anio)=2020) AND ((SubsidioCabezal.Mes)=7));


QUERY: Consulta5
  SQL: SELECT Imponible.CodEmpresa, Imponible.Concepto, Sum(Imponible.Importe) AS SumaDeImporte
FROM Imponible
WHERE (((Imponible.Mes)=1) AND ((Imponible.Anio)=2023))
GROUP BY Imponible.CodEmpresa, Imponible.Concepto;


QUERY: qCerti05
